scripts/pre-deploy-check.ts (‡∂î‡∂∫‡∑è ‡∂Ø‡∑ê‡∂±‡∂ß‡∂∏‡∂≠‡∑ä ‡∂≠‡∑í‡∂∫‡∑ô‡∂±‡∑Ä‡∑è):

import { execSync } from 'child_process';
import { statSync, readdirSync } from 'fs';
import { join } from 'path';

const MAX_SIZE_MB = 200; // Safety margin before 250MB
const BUILD_DIR = '.next/server';

function getFolderSize(dir: string): number {
  let size = 0;
  try {
    const files = readdirSync(dir);
    for (const file of files) {
      const filePath = join(dir, file);
      const stats = statSync(filePath);
      if (stats.isDirectory()) {
        size += getFolderSize(filePath);
      } else {
        size += stats.size;
      }
    }
  } catch (error) {
    console.error(`Error reading ${dir}:`, error);
  }
  return size;
}

async function checkBuildSize() {
  console.log('üîç Checking build size...\n');

  // Clean build
  console.log('üßπ Cleaning previous build...');
  execSync('rm -rf .next', { stdio: 'inherit' });

  // Generate Prisma
  console.log('üì¶ Generating Prisma client...');
  execSync('npx prisma generate', { stdio: 'inherit' });

  // Build
  console.log('üî® Building project...');
  execSync('npm run build', { stdio: 'inherit' });

  // Check sizes
  const serverSize = getFolderSize(BUILD_DIR);
  const sizeMB = (serverSize / (1024 * 1024)).toFixed(2);

  console.log('\nüìä Build Size Report:');
  console.log(`Server bundle: ${sizeMB} MB`);
  console.log(`Limit: ${MAX_SIZE_MB} MB`);

  if (parseFloat(sizeMB) > MAX_SIZE_MB) {
    console.error(`\n‚ùå Build size (${sizeMB}MB) exceeds ${MAX_SIZE_MB}MB!`);
    process.exit(1);
  }

  console.log('\n‚úÖ Build size check passed!');
}

checkBuildSize().catch((error) => {
  console.error('Error:', error);
  process.exit(1);
});
